(require 'package)

;; Repositorios en orden de prioridad
(setq package-archives
      '(("gnu" . "https://elpa.gnu.org/packages/")
        ("nongnu" . "https://elpa.nongnu.org/nongnu/")
        ("melpa" . "https://melpa.org/packages/")))

;; Priorizar GNU ELPA para paquetes estables
(setq package-archive-priorities
      '(("gnu" . 10)
        ("nongnu" . 5)
        ("melpa" . 0)))

(package-initialize)

;; Refrescar contenido de paquetes (una sola vez si faltan)
(unless package-archive-contents
  (package-refresh-contents))

;; Instalar use-package si no existe (ya se refresco arriba)
(unless (package-installed-p 'use-package)
  (package-install 'use-package))

;; Configuración de use-package
(require 'use-package)
(setq use-package-always-ensure t)    ; Auto-descargar paquetes
(setq use-package-always-pin nil)     ; No fijar versiones (evita "Not found" en MELPA)
(setq use-package-verbose init-file-debug) ; Solo verbose en debug (emacs --debug-init)

;; Reintentar con refresh si un paquete no se encuentra en MELPA
(define-advice package-install (:around (orig-fn &rest args) refresh-on-fail)
  "Si falla la instalación, refrescar el índice de paquetes y reintentar."
  (condition-case _err
      (apply orig-fn args)
    (error
     (package-refresh-contents)
     (apply orig-fn args))))

;; Sincronizar PATH con la shell (Optimizado)
(use-package exec-path-from-shell
  :ensure t
  :if (memq window-system '(mac ns x pgtk)) ; Solo en modo gráfico
  :config
  ;; OPTIMIZACIÓN CRÍTICA:
  ;; En macOS (mac/ns), necesitamos "-l" "-i" para leer todas las variables.
  ;; En Linux (x/pgtk), usar "-i" es MUY LENTO. Lo quitamos aquí.
  (if (memq window-system '(mac ns))
      (setq exec-path-from-shell-arguments '("-l" "-i"))
    ;; En Linux, por defecto no usamos argumentos pesados, o solo "-l" si es necesario
    (setq exec-path-from-shell-arguments '("-l")))

  (exec-path-from-shell-initialize))

(defun my/git-init-and-commit (project-dir &optional project-name)
  "Inicializa git en PROJECT-DIR y hace el primer commit con todo el scaffolding.
   PROJECT-NAME es opcional, se usa en el mensaje del commit."
  (let ((name (or project-name
                  (file-name-nondirectory (directory-file-name project-dir))))
        (default-directory project-dir))
    (if (not (executable-find "git"))
        (message "⚠️ Git no encontrado en el PATH")
      (let ((result (shell-command-to-string
                     (format
                      "git init && git add -A && git commit -m \"init: scaffolding inicial de %s\""
                      (shell-quote-argument name)))))
        (if (string-match-p "master\\|main" result)
            (message "✅ Git inicializado y primer commit listo en: %s" project-dir)
          (message "⚠️ Algo falló con git: %s" result))))))

;; --- Hydra de Compilación Universal (C-c c) ---
;; Alternativa a F5/F7/F12 — funciona en Termux y en terminal sin F-keys
(defun my/stop-server ()
  "Para el proceso corriendo en *compilation* o *live-server-log*.
Si hay un proceso de compilación activo, lo mata."
  (interactive)
  (let ((stopped nil))
    ;; Matar proceso de live-server si existe
    (when (get-process "live-server-proc")
      (ignore-errors (kill-process (get-process "live-server-proc")))
      (message "🛑 Live server detenido")
      (setq stopped t))
    ;; Matar proceso de *compilation* si existe
    (when (get-buffer-process "*compilation*")
      (ignore-errors (kill-process (get-buffer-process "*compilation*")))
      (message "🛑 Proceso de compilación detenido")
      (setq stopped t))
    ;; Matar proceso de vterm si existe
    (when (and (get-buffer "*vterm*") (get-buffer-process "*vterm*"))
      (message "ℹ️ Para detener un proceso en vterm, usa C-c C-c en el buffer"))
    (unless stopped
      (message "⚠️ No hay ningún servidor o proceso corriendo"))))

(defun my/new-c-project (name)
  "Crea proyecto C con Makefile profesional (build/test/coverage/asan/valgrind)."
  (interactive "sNombre del proyecto: ")
  (let ((project-dir (expand-file-name name (concat (my/mu-projects-base) "c/"))))

    (make-directory (concat project-dir "/include")      t)
    (make-directory (concat project-dir "/build")        t)
    (make-directory (concat project-dir "/tests/unity")  t)

    ;; Makefile profesional
    (with-temp-file (concat project-dir "/Makefile")
      (insert (format "PROJECT = %s\n" name))
      (insert "CC      = gcc\n")
      (insert "CFLAGS  = -Wall -Wextra -std=c11 -g -Iinclude\n\n")
      (insert "SRC     = $(wildcard src/*.c)\n")
      (insert "OBJ     = $(patsubst src/%.c, build/%.o, $(SRC))\n")
      (insert "TARGET  = build/$(PROJECT)\n\n")
      (insert "UNITY   = tests/unity/unity.c\n")
      (insert "TESTS   = $(wildcard tests/test_*.c)\n\n")
      (insert ".PHONY: all build test coverage asan ubsan valgrind clean\n\n")
      (insert "all: build\n\n")
      (insert "build: $(TARGET)\n\n")
      (insert "$(TARGET): $(OBJ) | build/\n")
      (insert "\t$(CC) $(CFLAGS) -o $@ $^\n\n")
      (insert "build/%.o: src/%.c | build/\n")
      (insert "\t$(CC) $(CFLAGS) -c -o $@ $<\n\n")
      (insert "build/:\n\t@mkdir -p $@\n\n")
      (insert "# Tests con Unity (descarga automática en primer uso)\n")
      (insert "test: $(UNITY) build\n")
      (insert "\t$(CC) $(CFLAGS) $(TESTS) $(UNITY) $(filter-out build/main.o,$(OBJ)) -o build/test_runner\n")
      (insert "\t@echo '=== Tests ==='\n")
      (insert "\t./build/test_runner\n\n")
      (insert "$(UNITY):\n")
      (insert "\t@echo 'Descargando Unity...'\n")
      (insert "\t@mkdir -p tests/unity\n")
      (insert "\t@curl -sLo tests/unity/unity.h     https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity.h\n")
      (insert "\t@curl -sLo tests/unity/unity.c     https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity.c\n")
      (insert "\t@curl -sLo tests/unity/unity_internals.h https://raw.githubusercontent.com/ThrowTheSwitch/Unity/master/src/unity_internals.h\n\n")
      (insert "# Cobertura (requiere gcc, gcov, lcov, genhtml)\n")
      (insert "coverage: CFLAGS += --coverage\n")
      (insert "coverage: clean build test\n")
      (insert "\t@mkdir -p coverage\n")
      (insert "\tlcov -c -d build/ -o coverage/coverage.info --rc lcov_branch_coverage=1\n")
      (insert "\tgenhtml coverage/coverage.info -o coverage/\n")
      (insert "\t@echo '=== Reporte en coverage/index.html ==='\n\n")
      (insert "# AddressSanitizer + UBSanitizer\n")
      (insert "asan: CFLAGS += -fsanitize=address,undefined\n")
      (insert "asan: clean $(TARGET)\n")
      (insert "\t./$(TARGET)\n\n")
      (insert "# Solo UBSanitizer\n")
      (insert "ubsan: CFLAGS += -fsanitize=undefined\n")
      (insert "ubsan: clean $(TARGET)\n")
      (insert "\t./$(TARGET)\n\n")
      (insert "# Ejecutar\n")
      (insert "run: build\n")
      (insert "\t./$(TARGET)\n\n")
      (insert "# Valgrind\n")
      (insert "valgrind: build\n")
      (insert "\tvalgrind --leak-check=full --show-leak-kinds=all --track-origins=yes ./$(TARGET)\n\n")
      (insert "clean:\n")
      (insert "\trm -rf build/ coverage/ *.gcov *.gcda *.gcno\n"))

    ;; compile_flags.txt para clangd
    (with-temp-file (concat project-dir "/compile_flags.txt")
      (insert "-Iinclude\n-Wall\n-Wextra\n-std=c11\n-g\n"))

    ;; main.c
    (with-temp-file (concat project-dir "/src/main.c")
      (insert "#include <stdio.h>\n\n")
      (insert "int main(void) {\n")
      (insert "    printf(\"Hola desde ")
      (insert name)
      (insert "\\n\");\n")
      (insert "    return 0;\n")
      (insert "}\n"))

    ;; Test de ejemplo
    (with-temp-file (concat project-dir "/tests/test_main.c")
      (insert "#include \"unity/unity.h\"\n")
      (insert (format "// #include \"../%s.h\"\n\n" name))
      (insert "void setUp(void)    {}\n")
      (insert "void tearDown(void) {}\n\n")
      (insert "void test_ejemplo(void) {\n")
      (insert "    TEST_ASSERT_EQUAL(2, 1 + 1);\n")
      (insert "}\n\n")
      (insert "int main(void) {\n")
      (insert "    UNITY_BEGIN();\n")
      (insert "    RUN_TEST(test_ejemplo);\n")
      (insert "    return UNITY_END();\n")
      (insert "}\n"))

    ;; .gitignore
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "build/\ncoverage/\ntests/unity/\n*.o\n*.gcov\n*.gcda\n*.gcno\n"))

    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "Proyecto C creado: %s" project-dir)
    (find-file (concat project-dir "/src/main.c"))))

;; --- Función Auxiliar: Divide por comas preservando comillas ---
(defun my/split-csv-preserve-quotes (text)
  "Divide TEXT por comas, ignorando las comas dentro de comillas dobles.
   Retorna una lista de elementos limpios."
  (let ((items '())
        (current-item "")
        (in-quote nil)
        (len (length text))
        (i 0))
    (while (< i len)
      (let ((char (aref text i)))
        (cond
         ((char-equal char ?\")
          (setq in-quote (not in-quote))
          (setq current-item (concat current-item (string char))))
         ((and (char-equal char ?,) (not in-quote))
          ;; Si es coma y NO estamos en comillas, guardar item
          (push (string-trim current-item) items)
          (setq current-item ""))
         (t
          (setq current-item (concat current-item (string char))))))
      (setq i (1+ i)))
    ;; Agregar el último elemento
    (push (string-trim current-item) items)
    (nreverse items)))

;; --- Función Principal: Smart Reverse ---
(defun my/smart-reverse ()
  "Rota, transpone o invierte listas CSV de forma inteligente.
   1. Selección: Mueve lo seleccionado al inicio.
   2. Cursor en medio: Transpone el elemento actual con el siguiente.
   3. Cursor al final: Invierte toda la línea."
  (interactive)
  (let* ((separator ", ")
         (eol (line-end-position))
         (bol (line-beginning-position))
         (full-text (buffer-substring-no-properties bol eol))
         (items (my/split-csv-preserve-quotes full-text))
         (new-text nil))

    (cond
     ;; --- CASO 1: Región Seleccionada (Rotar) ---
     ((use-region-p)
      (let* ((beg (region-beginning))
             (end (region-end))
             (sel-text (buffer-substring-no-properties beg end))
             (sel-items (my/split-csv-preserve-quotes sel-text))
             (all-items (my/split-csv-preserve-quotes full-text))
             ;; Filtrar items que NO están en la selección
             (rest-items (cl-remove-if (lambda (x) (member x sel-items)) all-items)))
        (setq new-text (string-join (append sel-items rest-items) separator))))

     ;; --- CASO 2: Cursor al final (Invertir Todo) ---
     ((>= (point) (1- eol))
      (setq new-text (string-join (nreverse items) separator)))

     ;; --- CASO 3: Transponer (Intercambiar vecino) ---
     (t
      (let ((current-len 0)
            (found-index -1)
            (cursor-rel (- (point) bol))) ;; Posición relativa al inicio de línea

        ;; Buscar en qué índice cae el cursor sumando longitudes
        (dotimes (i (length items))
          (let ((item-len (length (nth i items))))
            ;; Sumamos largo del item + 2 (por la coma y espacio)
            (setq current-len (+ current-len item-len 2))
            ;; Si aún no encontramos y el cursor está antes de este límite
            (when (and (< found-index 0) (<= cursor-rel current-len))
              (setq found-index i))))

        ;; Si no encontró (ej: cursor muy al final), asume el último
        (when (< found-index 0) (setq found-index (1- (length items))))

        ;; Lógica de Transposición
        (if (and (>= found-index 0) (< found-index (1- (length items))))
            ;; Si hay un siguiente elemento, intercambiamos A <-> B
            (let ((elem-a (nth found-index items))
                  (elem-b (nth (1+ found-index) items)))
              (let ((rotated-items (append (cl-subseq items 0 found-index)
                                           (list elem-b elem-a)
                                           (cl-subseq items (+ found-index 2)))))
                (setq new-text (string-join rotated-items separator))))

          ;; Si estamos en el último elemento (sin siguiente), invertimos todo por defecto
          (setq new-text (string-join (reverse items) separator))))))

    ;; --- Aplicar ---
    (if new-text
        (progn
          (delete-region bol eol)
          (insert new-text))
      (message "No se pudo procesar la lista."))))

(defun my/forja-warn-missing (module binary
