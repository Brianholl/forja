* OpenClaw — Agente Autónomo Completo

OpenClaw es un agente IA completo (Node.js, ~1.5GB RAM) con ecosistema
de skills (5400+ en ClawHub), browser automation, memoria persistente
y canales de mensajería (Telegram, Discord, etc.).

Solo disponible en perfil CASA (requiere más recursos que PicoClaw).

* 2. Directorio de trabajo por alumno

#+begin_src emacs-lisp
(defun my/openclaw-workspace ()
  "Retorna el directorio de OpenClaw del alumno activo."
  (if (bound-and-true-p my/mu-alumno-actual)
      (expand-file-name
       (concat my/mu-alumno-actual "/.openclaw/")
       my/mu-base-dir)
    (expand-file-name ".openclaw/" "~/")))

(defun my/openclaw-ensure-workspace ()
  "Crea la estructura de directorios de OpenClaw si no existe."
  (let ((ws (my/openclaw-workspace)))
    (dolist (subdir '("agents/forja" "memory" "skills"))
      (make-directory (expand-file-name subdir ws) t))
    ;; Crear SOUL.md (identidad del agente)
    (let ((soul-file (expand-file-name "agents/forja/SOUL.md" ws)))
      (unless (file-exists-p soul-file)
        (with-temp-file soul-file
          (insert "# FORJA Agent

Sos un asistente técnico para el entorno de desarrollo FORJA.
Tu rol es ayudar a alumnos y docentes a resolver problemas técnicos.

## Principios
- Explicá cada paso que hacés para que el alumno aprenda
- Siempre mostrá el diff antes de aplicar un cambio
- Nunca borres archivos sin confirmación
- Preferí soluciones simples sobre complejas
- Respondé en español

## Herramientas disponibles

- Archivos: podés leer y escribir archivos en el workspace

* 3. Control del proceso

#+begin_src emacs-lisp
(defun my/openclaw-running-p ()
  "Verifica si OpenClaw está corriendo."
  (and my/openclaw-process
       (process-live-p my/openclaw-process)))

(defun my/openclaw-start ()
  "Inicia OpenClaw con el workspace del alumno activo."
  (interactive)
  (if (my/openclaw-running-p)
      (message "🦞 OpenClaw ya está corriendo")
    (unless (file-executable-p my/openclaw-bin)
      (user-error "OpenClaw no encontrado en %s. Ejecuta install.sh --perfil casa" my/openclaw-bin))
    (let* ((ws (my/openclaw-ensure-workspace))
           (agent-dir (expand-file-name "agents/forja/" ws))
           (default-directory ws)
           (process-environment
            (append
             (list (format "OPENCLAW_AGENT_DIR=%s" agent-dir)
                   "OLLAMA_API_KEY=ollama-local")
             process-environment)))
      (setq my/openclaw-process
            (start-process "openclaw" "*openclaw*"
                           my/openclaw-bin "gateway"
                           "--port" (number-to-string my/openclaw-port)))
      (set-process-query-on-exit-flag my/openclaw-process nil)
      (message "🦞 OpenClaw iniciado (workspace: %s, puerto: %d)"
               ws my/openclaw-port))))

(defun my/openclaw-stop ()
  "Detiene OpenClaw."
  (interactive)
  (if (my/openclaw-running-p)
      (progn
        (kill-process my/openclaw-process)
        (setq my/openclaw-process nil)
        (message "🛑 OpenClaw detenido"))
    (message "OpenClaw no está corriendo")))

(defun my/openclaw-status ()
  "Muestra el estado de OpenClaw."
  (interactive)
  (message "OpenClaw: %s | Workspace: %s | Puerto: %d | Modelo: %s"
           (if (my/openclaw-running-p) "corriendo" "detenido")
           (my/openclaw-workspace)
           my/openclaw-port
           my/openclaw-default-model))

(defun my/openclaw-open-logs ()
  "Abre el buffer de logs de OpenClaw."
  (interactive)
  (if (get-buffer "*openclaw*")
      (switch-to-buffer-other-window "*openclaw*")
    (message "No hay logs de OpenClaw (no está corriendo)")))

(defun my/openclaw-open-web ()
  "Abre la interfaz web de OpenClaw en el navegador."
  (interactive)
  (if (my/openclaw-running-p)
      (browse-url (format "http://localhost:%d" my/openclaw-port))
    (user-error "OpenClaw no está corriendo. Usa C-c U O s para iniciarlo")))
#+end_src

* 4. Enviar tareas al agente

#+begin_src emacs-lisp
(defun my/openclaw-send (message)
  "Envía un mensaje/tarea a OpenClaw via CLI."
  (interactive "sTarea para OpenClaw: ")
  (unless (file-executable-p my/openclaw-bin)
    (user-error "OpenClaw no encontrado"))
  (let* ((ws (my/openclaw-workspace))
         (output-buf "*openclaw-output*")
         (cmd (format "OPENCLAW_AGENT_DIR=%s/agents/forja %s send '%s'"
                      ws my/openclaw-bin
                      (shell-quote-argument message))))
    (async-shell-command cmd output-buf)
    (message "🦞 Tarea enviada a OpenClaw")))

(defun my/openclaw-resolve-ticket ()
  "Lee el ticket bajo el cursor y pide a OpenClaw que lo analice."
  (interactive)
  (unless (derived-mode-p 'org-mode)
    (user-error "Este comando funciona en archivos .org"))
  (let* ((body (save-excursion
                 (org-back-to-heading t)
                 (let ((start (point))
                       (end (org-end-of-subtree t)))
                   (buffer-substring-no-properties start end))))
         (prompt (format "Analiza este ticket de soporte y propone una solución detallada paso a paso:\n\n%s" body)))
    (my/openclaw-send prompt)))

(defun my/openclaw-diagnose-project ()
  "Pide a OpenClaw que diagnostique el proyecto actual."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root)
                                (projectile-project-root))
                           default-directory))
         (prompt (format "Diagnostica el proyecto en %s. Ejecuta el build, analiza errores, revisa el código y sugiere correcciones con diffs." project-root)))
    (my/openclaw-send prompt)))

(defun my/openclaw-review-buffer ()
  "Pide a OpenClaw que revise el archivo actual en profundidad."
  (interactive)
  (let* ((file (buffer-file-name))
         (prompt (format "Hacé un code review completo del archivo %s. Buscá bugs, vulnerabilidades de seguridad, problemas de rendimiento y sugiere mejoras. Mostrá los diffs propuestos." file)))
    (if file
        (my/openclaw-send prompt)
      (message "❌ Este buffer no tiene archivo asociado"))))

(defun my/openclaw-generate-tests ()
  "Pide a OpenClaw que genere tests para el archivo actual."
  (interactive)
  (let* ((file (buffer-file-name))
         (prompt (format "Lee el archivo %s, identifica las funciones principales y generá un archivo de tests completo con casos límite. Escribí el archivo de tests al lado del original." file)))
    (if file
        (my/openclaw-send prompt)
      (message "❌ Este buffer no tiene archivo asociado"))))
#+end_src
