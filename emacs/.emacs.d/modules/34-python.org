* 40. Virtualenv

#+begin_src emacs-lisp
(defun my/python-find-venv ()
  "Busca un virtualenv en el proyecto actual.
Retorna la ruta al directorio del venv, o nil."
  (let* ((root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                   (locate-dominating-file default-directory "requirements.txt")
                   default-directory))
         (venv-dirs '("venv" ".venv" "env")))
    (catch 'found
      (dolist (dir venv-dirs)
        (let ((path (expand-file-name dir root)))
          (when (file-directory-p path)
            (throw 'found path))))
      nil)))

(defun my/python-activate-venv ()
  "Activa el virtualenv del proyecto actual si existe."
  (interactive)
  (let ((venv (my/python-find-venv)))
    (if venv
        (progn
          (setenv "VIRTUAL_ENV" venv)
          (setenv "PATH" (concat venv "/bin:" (getenv "PATH")))
          (setq python-shell-interpreter (concat venv "/bin/python"))
          (message "🐍 Virtualenv activado: %s" venv))
      (message "⚠️ No se encontró virtualenv (venv/.venv/env)"))))
#+end_src

* 50. Compilar y Ejecutar

#+begin_src emacs-lisp
(defun my/python-run ()
  "Ejecuta el archivo Python actual.
Detecta FastAPI (uvicorn), Django (manage.py) o ejecuta el archivo directamente."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           (locate-dominating-file default-directory "requirements.txt")
                           default-directory))
         (default-directory project-root)
         (manage-py (expand-file-name "manage.py" project-root))
         (main-py (expand-file-name "main.py" project-root))
         (venv (my/python-find-venv))
         ;; En Termux usar siempre python del sistema (venv problemático)
         (python (cond
                  ((bound-and-true-p my/is-termux) "python")
                  (venv (concat venv "/bin/python"))
                  (t "python"))))
    (cond
     ;; FastAPI: detecta uvicorn en requirements.txt o main.py con FastAPI
     ((and (file-exists-p main-py)
           (with-temp-buffer
             (insert-file-contents main-py)
             (search-forward "FastAPI" nil t)))
      (compile (format "%s -m uvicorn main:app --reload --host 0.0.0.0" python)))
     ;; Django
     ((file-exists-p manage-py)
      (compile (format "%s manage.py runserver" python)))
     ;; Archivo suelto
     ((buffer-file-name)
      (save-buffer)
      (compile (format "%s %s" python (file-name-nondirectory buffer-file-name))))
     (t (message "❌ No se encontró archivo Python para ejecutar")))))

(defun my/python-test ()
  "Ejecuta tests del proyecto Python.
Detecta pytest, Django test o unittest."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           default-directory))
         (default-directory project-root)
         (venv (my/python-find-venv))
         (python (if venv (concat venv "/bin/python") "python")))
    (cond
     ;; pytest
     ((or (file-exists-p (expand-file-name "pytest.ini" project-root))
          (file-exists-p (expand-file-name "pyproject.toml" project-root)))
      (compile (format "%s -m pytest -v" python)))
     ;; Django
     ((file-exists-p (expand-file-name "manage.py" project-root))
      (compile (format "%s manage.py test" python)))
     ;; unittest fallback
     (t (compile (format "%s -m unittest discover -v" python))))))

(defun my/python-keybindings ()
  "Atajos locales para desarrollo en Python."
  (local-set-key (kbd "<f5>") #'my/python-run)
  (local-set-key (kbd "C-c C-t") #'my/python-test)
  (local-set-key (kbd "C-c C-v") #'my/python-activate-venv))

(add-hook 'python-ts-mode-hook #'my/python-keybindings)
(add-hook 'python-mode-hook    #'my/python-keybindings)
#+end_src

(defun my/new-python-project (name)
  "Crea un nuevo proyecto Python con FastAPI.
En PC crea virtualenv; en Termux usa Python del sistema."
  (interactive "sNombre del proyecto: ")
  (let* ((project-dir (expand-file-name name (concat (my/mu-projects-base) "python/")))
         (envrc-mode nil)
         (process-environment (cons "DIRENV_DIR=" process-environment)))
    (make-directory project-dir t)
    (with-temp-file (concat project-dir "/main.py")
      (insert "from fastapi import FastAPI\n\n")
      (insert "app = FastAPI()\n\n\n")
      (insert "@app.get(\"/\")\n")
      (insert "def root():\n")
      (insert (format "    return {\"message\": \"Hola desde %s\"}\n\n\n" name))
      (insert "@app.get(\"/health\")\n")
      (insert "def health():\n")
      (insert "    return {\"status\": \"ok\"}\n"))
    (with-temp-file (concat project-dir "/requirements.txt")
      (insert "fastapi\nuvicorn[standard]\npython-dotenv\n"))
    (with-temp-file (concat project-dir "/.env")
      (insert "DEBUG=true\nPORT=8000\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "__pycache__/\n*.pyc\nvenv/\n.venv/\n.env\n*.egg-info/\ndist/\nbuild/\n"))
    (let ((default-directory project-dir))
      (if (bound-and-true-p my/is-termux)
          (async-shell-command "pip install fastapi uvicorn python-dotenv" "*pip-install*")
        (when (= 0 (shell-command "python -m venv venv"))
          (async-shell-command "venv/bin/pip install -r requirements.txt" "*venv-install*"))))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "🐍 Proyecto Python (FastAPI) creado: %s" project-dir)
    (find-file (concat project-dir "/main.py"))))
#+end_src
