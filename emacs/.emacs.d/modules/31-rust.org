  :ensure nil

* 40. Compilar y Ejecutar

#+begin_src emacs-lisp
(defun my/rust-run ()
  "Compila y ejecuta el proyecto Rust actual con cargo run."
  (interactive)
  (let* ((project-root
          (or (and (fboundp 'projectile-project-root) (projectile-project-root))
              (locate-dominating-file default-directory "Cargo.toml")
              default-directory))
         (default-directory project-root))
    (if (file-exists-p (expand-file-name "Cargo.toml" project-root))
        (compile "cargo run")
      (message "❌ No se encontró Cargo.toml. ¿Estás en un proyecto Rust?"))))

(defun my/rust-build ()
  "Compila el proyecto Rust sin ejecutar."
  (interactive)
  (let* ((project-root
          (or (and (fboundp 'projectile-project-root) (projectile-project-root))
              (locate-dominating-file default-directory "Cargo.toml")
              default-directory))
         (default-directory project-root))
    (if (file-exists-p (expand-file-name "Cargo.toml" project-root))
        (compile "cargo build")
      (message "❌ No se encontró Cargo.toml."))))

(defun my/rust-test ()
  "Ejecuta tests del proyecto."
  (interactive)
  (let ((default-directory
         (or (and (fboundp 'projectile-project-root) (projectile-project-root))
             (locate-dominating-file default-directory "Cargo.toml")
             default-directory)))
    (compile "cargo test")))

(defun my/debug-rust ()
  "Compila con debug symbols y lanza GDB."
  (interactive)
  (let* ((project-root (projectile-project-root))
         (target-dir (concat project-root "target/debug/"))
         (exe-name (file-name-nondirectory (directory-file-name project-root))))
    (compile "cargo build")
    (sit-for 2)
    (gdb (format "rust-gdb -i=mi %s%s" target-dir exe-name))))

(defun my/rust-keybindings ()
  "Atajos locales para desarrollo en Rust."
  (local-set-key (kbd "<f5>") #'my/rust-run)
  (local-set-key (kbd "C-c C-b") #'my/rust-build)
  (local-set-key (kbd "C-c C-t") #'my/rust-test)
  (local-set-key (kbd "C-c d") #'my/debug-rust))

(add-hook 'rust-mode-hook    #'my/rust-keybindings)
(add-hook 'rust-ts-mode-hook #'my/rust-keybindings)
#+end_src

#+begin_src emacs-lisp
(defun my/new-rust-project (name)
  "Crea un nuevo proyecto Rust con cargo."
  (interactive "sNombre del proyecto: ")
  (unless (executable-find "cargo")
    (user-error "cargo no encontrado — instala Rust con: curl https://sh.rustup.rs -sSf | sh"))
  (let ((base-dir (concat (my/mu-projects-base) "rust/"))
        (project-dir (expand-file-name name (concat (my/mu-projects-base) "rust/"))))
    (make-directory base-dir t)
    (let ((default-directory base-dir))
      (shell-command (format "cargo new %s" name)))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "🦀 Proyecto Rust creado: %s" project-dir)
    (find-file (concat project-dir "/src/main.rs"))))

(defun my/new-rust-api-project (name)
  "Crea un nuevo proyecto Rust con Actix-web para REST API."
  (interactive "sNombre del proyecto: ")
  (let* ((base-dir (concat (my/mu-projects-base) "rust/"))
         (project-dir (expand-file-name name base-dir)))
    (make-directory base-dir t)
    (let ((default-directory base-dir))
      (shell-command (format "cargo new %s" name)))
    (with-temp-file (concat project-dir "/Cargo.toml")
      (insert "[package]\n")
      (insert (format "name = \"%s\"\n" name))
      (insert "version = \"0.1.0\"\n")
      (insert "edition = \"2021\"\n\n")
      (insert "[dependencies]\n")
      (insert "actix-web = \"4\"\n")
      (insert "actix-cors = \"0.7\"\n")
      (insert "serde = { version = \"1\", features = [\"derive\"] }\n")
      (insert "serde_json = \"1\"\n")
      (insert "dotenv = \"0.15\"\n")
      (insert "tokio = { version = \"1\", features = [\"full\"] }\n")
      (insert "env_logger = \"0.11\"\n")
      (insert "log = \"0.4\"\n"))
    (with-temp-file (concat project-dir "/src/main.rs")
      (insert "use actix_cors::Cors;\n")
      (insert "use actix_web::{get, post, web, App, HttpResponse, HttpServer, Responder};\n")
      (insert "use serde::{Deserialize, Serialize};\n\n")
      (insert "#[derive(Serialize)]\n")
      (insert "struct ApiResponse {\n    message: String,\n}\n\n")
      (insert "#[derive(Deserialize)]\n")
      (insert "struct CreateItem {\n    name: String,\n}\n\n")
      (insert "#[get(\"/\")]\n")
      (insert "async fn index() -> impl Responder {\n")
      (insert (format "    HttpResponse::Ok().json(ApiResponse {\n        message: \"Hola desde %s\".to_string(),\n    })\n" name))
      (insert "}\n\n")
      (insert "#[get(\"/health\")]\n")
      (insert "async fn health() -> impl Responder {\n")
      (insert "    HttpResponse::Ok().json(serde_json::json!({\"status\": \"ok\"}))\n}\n\n")
      (insert "#[get(\"/api/items\")]\n")
      (insert "async fn get_items() -> impl Responder {\n")
      (insert "    HttpResponse::Ok().json(serde_json::json!({\"items\": []}))\n}\n\n")
      (insert "#[post(\"/api/items\")]\n")
      (insert "async fn create_item(item: web::Json<CreateItem>) -> impl Responder {\n")
      (insert "    HttpResponse::Created().json(serde_json::json!({\n")
      (insert "        \"message\": format!(\"Item {} creado\", item.name)\n")
      (insert "    }))\n}\n\n")
      (insert "#[actix_web::main]\n")
      (insert "async fn main() -> std::io::Result<()> {\n")
      (insert "    dotenv::dotenv().ok();\n    env_logger::init();\n\n")
      (insert "    let port = std::env::var(\"PORT\").unwrap_or_else(|_| \"8080\".to_string());\n")
      (insert "    let addr = format!(\"0.0.0.0:{}\", port);\n\n")
      (insert "    println!(\"🦀 Server corriendo en http://localhost:{}\", port);\n\n")
      (insert "    HttpServer::new(|| {\n")
      (insert "        let cors = Cors::permissive();\n")
      (insert "        App::new().wrap(cors)\n")
      (insert "            .service(index).service(health)\n")
      (insert "            .service(get_items).service(create_item)\n")
      (insert "    })\n    .bind(&addr)?.run().await\n}\n"))
    (with-temp-file (concat project-dir "/.env")
      (insert "# Variables de entorno\nPORT=8080\nRUST_LOG=info\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "/target\n.env\n*.log\n"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "🦀 Proyecto Rust API (Actix-web) creado: %s" project-dir)
    (find-file (concat project-dir "/src/main.rs"))))
#+end_src
