* 20. PHP Mode

~php-ts-mode~ de Emacs 30 usa queries con ~#match~ que fallan en ejecución con
tree-sitter 0.26.x (el error aparece en ~jit-lock~, no en compilación).
Se usa ~php-mode~ hasta que Emacs corrija sus queries a la sintaxis ~#match?~.

#+begin_src emacs-lisp
;; php-ts-mode de Emacs 30 usa queries con #match que fallan en ejecución

  :config

* 40. Formateo con Prettier

#+begin_src emacs-lisp
(defun my/php-format-buffer ()
  "Formatea el buffer PHP con prettier (si disponible) o lsp."
  (interactive)
  (if (executable-find "prettier")
      (let ((file (buffer-file-name)))
        (when file
          (save-buffer)
          (shell-command (format "prettier --write %s" (shell-quote-argument file)))
          (revert-buffer t t t)
          (message "✅ Formateado con prettier")))
    (lsp-format-buffer)))
#+end_src

* 50. Compilar y Ejecutar

#+begin_src emacs-lisp
(defun my/php-run ()
  "Ejecuta el archivo PHP actual.
Detecta Laravel (artisan serve), o ejecuta el archivo directamente."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           (locate-dominating-file default-directory "composer.json")
                           default-directory))
         (default-directory project-root)
         (artisan (expand-file-name "artisan" project-root)))
    (cond
     ;; Laravel: php artisan serve
     ((file-exists-p artisan)
      (compile "php artisan serve"))
     ;; PHP built-in server desde public/ o directorio actual
     ((file-directory-p (expand-file-name "public" project-root))
      (compile (format "php -S localhost:8000 -t %s"
                       (expand-file-name "public" project-root))))
     ;; Archivo suelto
     ((buffer-file-name)
      (save-buffer)
      (compile (format "php %s" (file-name-nondirectory buffer-file-name))))
     (t (message "❌ No se encontró archivo PHP para ejecutar")))))

(defun my/php-test ()
  "Ejecuta tests del proyecto PHP.
Detecta Laravel (artisan test) o PHPUnit."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           (locate-dominating-file default-directory "composer.json")
                           default-directory))
         (default-directory project-root)
         (artisan (expand-file-name "artisan" project-root))
         (phpunit (expand-file-name "vendor/bin/phpunit" project-root)))
    (cond
     ;; Laravel
     ((file-exists-p artisan)
      (compile "php artisan test"))
     ;; PHPUnit directo
     ((file-exists-p phpunit)
      (compile "./vendor/bin/phpunit"))
     ;; Fallback
     (t (compile "php -r \"echo 'No se encontró framework de tests';\"")))))

(defun my/php-keybindings ()
  "Atajos locales para desarrollo en PHP."
  (local-set-key (kbd "<f5>") #'my/php-run)
  (local-set-key (kbd "C-c C-t") #'my/php-test)
  (local-set-key (kbd "C-c C-f") #'my/php-format-buffer))

(add-hook 'php-ts-mode-hook #'my/php-keybindings)
(add-hook 'php-mode-hook    #'my/php-keybindings)

(my/forja-register-lang
 :modes   '(php-mode php-ts-mode)
 :run     'my/php-run           :run-fallback    "php -S localhost:8000"
 :build   nil                   :build-fallback  "composer install"
 :test    'my/php-test          :test-fallback   "php artisan test"
 :format  'my/php-format-buffer :format-fallback nil)
#+end_src

* 60. Generadores de Proyecto

#+begin_src emacs-lisp
(defun my/new-php-project (name)
  "Crea un nuevo proyecto PHP.
Si composer está disponible, crea un proyecto Laravel.
Si no, crea estructura básica con index.php."
  (interactive "sNombre del proyecto: ")
  (let* ((base-dir (concat (my/mu-projects-base) "php/"))
         (project-dir (expand-file-name name base-dir)))
    (make-directory base-dir t)
    (if (executable-find "composer")
        (progn
          (message "Creando proyecto Laravel con composer (puede tardar)...")
          (let ((default-directory base-dir))
            (shell-command (format "composer create-project laravel/laravel %s" name)))
          (when (file-directory-p project-dir)
            (with-temp-file (concat project-dir "/.projectile") (insert ""))
            (message "🐘 Proyecto Laravel creado: %s" project-dir)
            (find-file (concat project-dir "/routes/web.php"))))
      (make-directory project-dir t)
      (make-directory (concat project-dir "/public") t)
      (make-directory (concat project-dir "/src") t)
      (with-temp-file (concat project-dir "/public/index.php")
        (insert "<?php\n\nrequire_once __DIR__ . '/../src/app.php';\n"))
      (with-temp-file (concat project-dir "/src/app.php")
        (insert "<?php\n\n")
        (insert "header('Content-Type: application/json');\n\n")
        (insert "$method = $_SERVER['REQUEST_METHOD'];\n")
        (insert "$path = parse_url($_SERVER['REQUEST_URI'], PHP_URL_PATH);\n\n")
        (insert "if ($path === '/' && $method === 'GET') {\n")
        (insert (format "    echo json_encode(['message' => 'Hola desde %s']);\n" name))
        (insert "} elseif ($path === '/health' && $method === 'GET') {\n")
        (insert "    echo json_encode(['status' => 'ok']);\n")
        (insert "} else {\n")
        (insert "    http_response_code(404);\n")
        (insert "    echo json_encode(['error' => 'Not found']);\n")
        (insert "}\n"))
      (with-temp-file (concat project-dir "/composer.json")
        (insert "{\n")
        (insert (format "    \"name\": \"brian/%s\",\n" name))
        (insert "    \"autoload\": {\"psr-4\": {\"App\\\\\\\\\": \"src/\"}},\n")
        (insert "    \"require\": {\"php\": \">=8.0\"}\n")
        (insert "}\n"))
      (with-temp-file (concat project-dir "/.gitignore")
        (insert "vendor/\n.env\ncomposer.lock\n"))
      (with-temp-file (concat project-dir "/.projectile") (insert ""))
      (my/git-init-and-commit project-dir name)
      (message "🐘 Proyecto PHP creado: %s" project-dir)
      (find-file (concat project-dir "/src/app.php")))))
#+end_src
