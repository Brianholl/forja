(defun my/new-java-project (name)
  "Crea un proyecto Java con Maven (estructura estándar + JUnit 5)."
  (interactive "sNombre del proyecto Java: ")
  (unless (executable-find "mvn")
    (user-error "Maven no encontrado — instala con: %s"
                (cond ((bound-and-true-p my/is-termux) "pkg install maven")
                      ((bound-and-true-p my/is-wsl)    "sudo apt install maven")
                      (t "sudo pacman -S maven"))))
  (let* ((project-dir (expand-file-name name (concat (my/mu-projects-base) "java/")))
         (main-dir    (concat project-dir "/src/main/java"))
         (test-dir    (concat project-dir "/src/test/java")))
    (make-directory main-dir t)
    (make-directory test-dir t)
    (with-temp-file (concat main-dir "/Main.java")
      (insert (format "public class Main {\n"))
      (insert "    public static void main(String[] args) {\n")
      (insert (format "        System.out.println(\"Hola desde %s!\");\n" name))
      (insert "    }\n")
      (insert "}\n"))
    (with-temp-file (concat test-dir "/MainTest.java")
      (insert "import org.junit.jupiter.api.Test;\n")
      (insert "import static org.junit.jupiter.api.Assertions.*;\n\n")
      (insert "class MainTest {\n")
      (insert "    @Test\n")
      (insert "    void testBasico() {\n")
      (insert "        assertEquals(2, 1 + 1);\n")
      (insert "    }\n")
      (insert "}\n"))
    (with-temp-file (concat project-dir "/pom.xml")
      (insert "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n")
      (insert "<project xmlns=\"http://maven.apache.org/POM/4.0.0\"\n")
      (insert "         xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"\n")

      (insert "  <modelVersion>4.0.0</modelVersion>\n\n")
      (insert "  <groupId>dev.forja</groupId>\n")
      (insert (format "  <artifactId>%s</artifactId>\n" (downcase name)))
      (insert "  <version>1.0-SNAPSHOT</version>\n")
      (insert "  <packaging>jar</packaging>\n\n")
      (insert "  <properties>\n")
      (insert "    <maven.compiler.source>17</maven.compiler.source>\n")
      (insert "    <maven.compiler.target>17</maven.compiler.target>\n")
      (insert "    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>\n")
      (insert "  </properties>\n\n")
      (insert "  <dependencies>\n")
      (insert "    <dependency>\n")
      (insert "      <groupId>org.junit.jupiter</groupId>\n")
      (insert "      <artifactId>junit-jupiter</artifactId>\n")
      (insert "      <version>5.10.2</version>\n")
      (insert "      <scope>test</scope>\n")
      (insert "    </dependency>\n")
      (insert "  </dependencies>\n\n")
      (insert "  <build>\n")
      (insert "    <plugins>\n")
      (insert "      <plugin>\n")
      (insert "        <groupId>org.apache.maven.plugins</groupId>\n")
      (insert "        <artifactId>maven-surefire-plugin</artifactId>\n")
      (insert "        <version>3.2.5</version>\n")
      (insert "      </plugin>\n")
      (insert "      <plugin>\n")
      (insert "        <groupId>org.codehaus.mojo</groupId>\n")
      (insert "        <artifactId>exec-maven-plugin</artifactId>\n")
      (insert "        <version>3.2.0</version>\n")
      (insert "        <configuration>\n")

* 50. Generador — Proyecto Kotlin (Gradle)

#+begin_src emacs-lisp
(defun my/new-kotlin-project (name)
  "Crea un proyecto Kotlin con Gradle (Kotlin DSL)."
  (interactive "sNombre del proyecto Kotlin: ")
  (unless (executable-find "gradle")
    (user-error "Gradle no encontrado — instala con: %s"
                (cond ((bound-and-true-p my/is-termux) "pkg install gradle")
                      ((bound-and-true-p my/is-wsl)    "sudo apt install gradle")
                      (t "sudo pacman -S gradle"))))
  (let* ((project-dir (expand-file-name name (concat (my/mu-projects-base) "kotlin/")))
         (src-dir     (concat project-dir "/src/main/kotlin")))
    (make-directory src-dir t)
    (with-temp-file (concat src-dir "/Main.kt")
      (insert "fun main() {\n")
      (insert (format "    println(\"Hola desde %s!\")\n" name))
      (insert "}\n"))
    (with-temp-file (concat project-dir "/build.gradle.kts")
      (insert "plugins {\n")
      (insert "    kotlin(\"jvm\") version \"2.0.21\"\n")
      (insert "    application\n")
      (insert "}\n\n")
      (insert "repositories {\n    mavenCentral()\n}\n\n")
      (insert "dependencies {\n")
      (insert "    testImplementation(kotlin(\"test\"))\n")
      (insert "}\n\n")
      (insert "application {\n")
      (insert "    mainClass = \"MainKt\"\n")
      (insert "}\n\n")
      (insert "tasks.test {\n    useJUnitPlatform()\n}\n"))
    (with-temp-file (concat project-dir "/settings.gradle.kts")
      (insert (format "rootProject.name = \"%s\"\n" (downcase name))))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert ".gradle/\nbuild/\n*.class\n"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "Proyecto Kotlin (Gradle) creado: %s" project-dir)
    (find-file (concat src-dir "/Main.kt"))))
#+end_src

* 60. Compilar / Ejecutar

#+begin_src emacs-lisp
(defun my/java-run ()
  "Compila y ejecuta el proyecto Java (mvn) o archivo suelto."
  (interactive)
  (let ((root (locate-dominating-file default-directory "pom.xml")))
    (if root
        (let ((default-directory root))
          (compile "mvn -q compile exec:java"))
      (compile (format "javac %s && java %s"
                       (buffer-file-name)
                       (file-name-sans-extension (buffer-file-name)))))))

(defun my/java-test ()
  "Ejecuta los tests del proyecto Maven."
  (interactive)
  (let* ((root (locate-dominating-file default-directory "pom.xml"))
         (default-directory (or root default-directory)))
    (compile "mvn -q test")))

(defun my/kotlin-run ()
  "Compila y ejecuta el proyecto Kotlin (Gradle)."
  (interactive)
  (let* ((root (locate-dominating-file default-directory "build.gradle.kts"))
         (default-directory (or root default-directory)))
    (if root
        (compile "gradle -q run")
      (message "No se encontró build.gradle.kts"))))

(defun my/java-keybindings ()
  (local-set-key (kbd "<f5>")    #'my/java-run)
  (local-set-key (kbd "C-c C-t") #'my/java-test))

(defun my/kotlin-keybindings ()
  (local-set-key (kbd "<f5>") #'my/kotlin-run))

(add-hook 'java-mode-hook      #'my/java-keybindings)
(add-hook 'java-ts-mode-hook   #'my/java-keybindings)
(add-hook 'kotlin-mode-hook    #'my/kotlin-keybindings)
(add-hook 'kotlin-ts-mode-hook #'my/kotlin-keybindings)
#+end_src

* 70. Dependencias y registro

#+begin_src emacs-lisp
(my/forja-dep :module "java"   :binary "java"   :type :required
              :arch "jdk17-openjdk" :wsl "default-jdk" :termux "openjdk-17")
(my/forja-dep :module "java"   :binary "mvn"    :type :required
              :arch "maven" :wsl "maven" :termux "maven")
(my/forja-dep :module "kotlin" :binary "gradle" :type :required
              :arch "gradle" :wsl "gradle" :termux "gradle")

(my/forja-register-lang
 :modes   '(java-mode java-ts-mode)
 :run     'my/java-run      :run-fallback    "mvn -q compile exec:java"
 :build   nil               :build-fallback  "mvn -q compile"
 :test    'my/java-test     :test-fallback   "mvn -q test"
 :format  'lsp-format-buffer :format-fallback nil)

(my/forja-register-lang
 :modes   '(kotlin-mode kotlin-ts-mode)
 :run     'my/kotlin-run    :run-fallback    "gradle -q run"
 :build   nil               :build-fallback  "gradle -q assemble"
 :test    nil               :test-fallback   "gradle -q test"
 :format  'lsp-format-buffer :format-fallback nil)
#+end_src
