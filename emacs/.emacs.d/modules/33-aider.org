      '("--model"          "openrouter/nvidia/nemotron-3-super-120b-a12b:free"

;; --- Minuet AI (Autocompletado predictivo tipo fantasma) ---
;; Usa qwen2.5-coder:0.5b offline en Ollama vía openai-fim-compatible
(use-package minuet
  :ensure t
  :config
  (setq minuet-provider 'openai-fim-compatible)
  (setq minuet-openai-fim-compatible-options
        '(:model "qwen2.5-coder:0.5b"
          :end-point "http://localhost:11434/v1/completions"
          :api-key "ollama"
          :name "Ollama"
          :optional (:max_tokens 128 :temperature 0.1)))

  (global-set-key (kbd "M-i") #'minuet-show-suggestion)
  (global-set-key (kbd "M-[") #'minuet-dismiss-suggestion)
  (define-key minuet-active-mode-map (kbd "TAB") #'minuet-accept-suggestion))
#+end_src

* 20. Funciones de Conveniencia

#+begin_src emacs-lisp
;; --- Verificar si Aider está corriendo ---
(defun my/aider-buffer-running-p ()
  "Verifica si hay un buffer de Aider activo."
  (let ((buf-name (aider-buffer-name)))
    (and buf-name
         (get-buffer buf-name)
         (comint-check-proc buf-name))))

(defun my/aider-open-project ()
  "Abre Aider en la raíz del proyecto actual."
  (interactive)
  (let ((default-directory
         (or (and (fboundp 'projectile-project-root)
                  (projectile-project-root))
             default-directory)))
    (aider-run-aider)
    (message "🤖 Aider iniciado en: %s" default-directory)))

(defun my/aider-add-current-file ()
  "Agrega el archivo actual al contexto de Aider.
   Si Aider no está corriendo, lo inicia primero."
  (interactive)
  (if (buffer-file-name)
      (progn
        (unless (my/aider-buffer-running-p)
          (my/aider-open-project)
          (sit-for 2))
        (aider-add-current-file))
    (message "❌ Este buffer no tiene archivo asociado")))

(defun my/aider-ask-region ()
  "Pregunta a Aider sobre la región seleccionada."
  (interactive)
  (if (use-region-p)
      (aider-send-block-or-region)
    (message "❌ No hay región seleccionada")))

(defun my/aider-fix-flycheck ()
  "Pide a Aider que corrija los errores de Flycheck del buffer actual."
  (interactive)
  (unless (my/aider-buffer-running-p)
    (my/aider-open-project)
    (sit-for 2))
  (my/aider-add-current-file)
  (aider-flycheck-fix-errors-in-scope))

(defun my/aider-keybindings ()
  "Atajos de Aider para modos de programación."
  ;; Sesión
  (local-set-key (kbd "C-c i o") #'my/aider-open-project)
  (local-set-key (kbd "C-c i a") #'my/aider-add-current-file)
  (local-set-key (kbd "C-c i x") #'aider-exit)
  (local-set-key (kbd "C-c i s") #'aider-switch-to-buffer)
  ;; Código
  (local-set-key (kbd "C-c i c") #'aider-function-or-region-change)
  (local-set-key (kbd "C-c i r") #'my/aider-ask-region)
  (local-set-key (kbd "C-c i q") #'aider-ask-question)
  (local-set-key (kbd "C-c i f") #'my/aider-fix-flycheck)
  ;; Tests y planificación
  (local-set-key (kbd "C-c i t") #'aider-write-unit-test)
  (local-set-key (kbd "C-c i p") #'aider-start-software-planning)
  ;; Utilidades
  (local-set-key (kbd "C-c i u") #'aider-undo-last-change)
  (local-set-key (kbd "C-c i h") #'aider-open-history))

(add-hook 'prog-mode-hook #'my/aider-keybindings)
;; Modos que NO heredan de prog-mode → se agregan explícitamente
(add-hook 'web-mode-hook      #'my/aider-keybindings)  ; JSX / Vue
(add-hook 'html-ts-mode-hook  #'my/aider-keybindings)  ; HTML (hereda de text-mode)
#+end_src
