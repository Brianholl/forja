* OpenCode — Asistente Agentico (OpenRouter)

Integración Emacs para los comandos *no-interactivos* de OpenCode
(=opencode run=). El TUI completo se lanza desde =~/forja/opencode.sh=.

Requiere =OPENROUTER_API_KEY= (configurado por =connect.sh=).
Activar en profile.conf: =FORJA_FEATURES="...,opencode"=

* 1. Variables

#+begin_src emacs-lisp
(defvar my/opencode-buf-out "*opencode-output*"
  "Buffer de salida para comandos opencode run.")

(defvar my/opencode-available
  (cond
   (my/is-termux (executable-find "proot-distro"))
   (t            (executable-find "opencode")))
  "Non-nil si OpenCode está disponible en esta plataforma.")

(defun my/opencode--guard ()
  "Error si OpenCode no está disponible o modo examen activo."
  (unless my/opencode-available
    (user-error "OpenCode no disponible. Instalá via install.sh o connect.sh"))
  (when (bound-and-true-p my/forja-exam-mode-active)
    (user-error "[MODO EXAMEN] OpenCode deshabilitado")))

(defun my/opencode--wrap-run (prompt)
  "Devuelve el comando opencode run con PROMPT según plataforma."
  (let ((api-key (or (getenv "OPENROUTER_API_KEY") "")))
    (if my/is-termux
        (format "OPENROUTER_API_KEY='%s' proot-distro login ubuntu -- opencode run %s"
                api-key (shell-quote-argument prompt))
      (format "opencode run %s" (shell-quote-argument prompt)))))
#+end_src

* 2. TUI completo — lanzar en terminal externa

Para sesiones interactivas usar =~/forja/opencode.sh= desde la terminal.
Esta función abre alacritty directamente desde Emacs.

#+begin_src emacs-lisp
(defun my/opencode-run ()
  "Lanza OpenCode TUI en alacritty (terminal externa)."
  (interactive)
  (my/opencode--guard)
  (let* ((root (or (and (fboundp 'projectile-project-root)
                        (ignore-errors (projectile-project-root)))
                   default-directory))
         (script (expand-file-name "~/forja/opencode.sh")))
    (cond
     (my/is-termux
      (message "Corré en Termux: %s/forja/opencode.sh %s" (getenv "HOME") root))
     ((file-executable-p script)
      (start-process "opencode" nil "alacritty"
                     "--working-directory" root
                     "-e" script root)
      (message "OpenCode abierto en alacritty — %s" root))
     (t
      (user-error "No se encontró ~/forja/opencode.sh o alacritty")))))
#+end_src

* 3. Consultas inline (opencode run)

Envían un prompt a =opencode run= y muestran el resultado en un panel.

#+begin_src emacs-lisp
(defun my/opencode--run-async (prompt)
  "Ejecuta opencode run PROMPT y muestra resultado en panel inferior."
  (let ((cmd (my/opencode--wrap-run prompt)))
    (with-current-buffer (get-buffer-create my/opencode-buf-out)
      (read-only-mode -1)
      (erase-buffer)
      (insert (format "=== OpenCode — %s ===\n\n"
                      (format-time-string "%H:%M:%S"))))
    (async-shell-command cmd my/opencode-buf-out)
    (display-buffer my/opencode-buf-out
                    '(display-buffer-in-side-window
                      (side . bottom) (window-height . 0.35)))))

(defun my/opencode-ask-file ()
  "Pregunta a OpenCode sobre el archivo actual."
  (interactive)
  (my/opencode--guard)
  (let* ((file (or (buffer-file-name) (user-error "Buffer sin archivo")))
         (q    (read-string "Pregunta: ")))
    (my/opencode--run-async (format "Archivo: %s\n\nPregunta: %s" file q))))

(defun my/opencode-ask-region ()
  "Pregunta a OpenCode sobre la región seleccionada."
  (interactive)
  (my/opencode--guard)
  (unless (use-region-p) (user-error "No hay región seleccionada"))
  (let* ((code (buffer-substring-no-properties (region-beginning) (region-end)))
         (lang (replace-regexp-in-string "-mode\\|-ts-mode" ""
                                         (symbol-name major-mode)))
         (q    (read-string "Pregunta: ")))
    (deactivate-mark)
    (my/opencode--run-async
     (format "```%s\n%s\n```\n\nPregunta: %s" lang code q))))

(defun my/opencode-process-file ()
  "Análisis agentico completo del archivo actual."
  (interactive)
  (my/opencode--guard)
  (let ((file (or (buffer-file-name) (user-error "Buffer sin archivo"))))
    (my/opencode--run-async
     (format "Analiza el archivo %s. Buscá bugs, problemas de lógica, performance y seguridad. Mostrá los cambios como diffs." file))
    (message "Procesando %s..." (file-name-nondirectory file))))

(defun my/opencode-generate-tests ()
  "Genera tests para el archivo actual."
  (interactive)
  (my/opencode--guard)
  (let ((file (or (buffer-file-name) (user-error "Buffer sin archivo"))))
    (my/opencode--run-async
     (format "Leé %s y generá un archivo de tests completo con casos normales y límite." file))))

(defun my/opencode-fix-errors ()
  "Pide a OpenCode que corrija los errores de Flycheck."
  (interactive)
  (my/opencode--guard)
  (let ((file (or (buffer-file-name) (user-error "Buffer sin archivo"))))
    (my/opencode--run-async
     (format "Archivo: %s\n\nCorregí todos los errores de linting y compilación." file))))
#+end_src

* 4. Keybindings

#+begin_src emacs-lisp
(defun my/opencode-keybindings ()
  "Atajos de OpenCode para modos de programación."
  (local-set-key (kbd "C-c O o") #'my/opencode-run)
  (local-set-key (kbd "C-c O p") #'my/opencode-ask-file)
  (local-set-key (kbd "C-c O r") #'my/opencode-ask-region)
  (local-set-key (kbd "C-c O f") #'my/opencode-process-file)
  (local-set-key (kbd "C-c O t") #'my/opencode-generate-tests)
  (local-set-key (kbd "C-c O e") #'my/opencode-fix-errors))

(add-hook 'prog-mode-hook #'my/opencode-keybindings)
(add-hook 'web-mode-hook      #'my/opencode-keybindings)
(add-hook 'html-ts-mode-hook  #'my/opencode-keybindings)
(global-set-key (kbd "C-c O o") #'my/opencode-run)
#+end_src
