(require 'package)

;; Repositorios en orden de prioridad
(setq package-archives
      '(("gnu" . "https://elpa.gnu.org/packages/")
        ("nongnu" . "https://elpa.nongnu.org/nongnu/")
        ("melpa" . "https://melpa.org/packages/")))

      '(("gnu" . 10)
        ("nongnu" . 5)
        ("melpa" . 0)))

;; Refrescar contenido de paquetes (una sola vez si faltan)
(unless package-archive-contents
  (package-refresh-contents))

;; Instalar use-package si no existe (ya se refresco arriba)
(unless (package-installed-p 'use-package)
  (package-install 'use-package))

;; Configuración de use-package
(require 'use-package)
(setq use-package-always-ensure t)    ; Auto-descargar paquetes
(setq use-package-always-pin nil)     ; No fijar versiones (evita "Not found" en MELPA)
(setq use-package-verbose init-file-debug) ; Solo verbose en debug (emacs --debug-init)

;; Reintentar con refresh si un paquete no se encuentra en MELPA
(define-advice package-install (:around (orig-fn &rest args) refresh-on-fail)
  "Si falla la instalación, refrescar el índice de paquetes y reintentar."
  (condition-case _err
      (apply orig-fn args)
    (error
     (package-refresh-contents)
     (apply orig-fn args))))

;; Sincronizar PATH con la shell (Optimizado)
(use-package exec-path-from-shell
  :ensure t
  :if (memq window-system '(mac ns x pgtk)) ; Solo en modo gráfico
  :config
  ;; OPTIMIZACIÓN CRÍTICA:
  ;; En macOS (mac/ns), necesitamos "-l" "-i" para leer todas las variables.
  ;; En Linux (x/pgtk), usar "-i" es MUY LENTO. Lo quitamos aquí.
  (if (memq window-system '(mac ns))
      (setq exec-path-from-shell-arguments '("-l" "-i"))
    ;; En Linux, por defecto no usamos argumentos pesados, o solo "-l" si es necesario
    (setq exec-path-from-shell-arguments '("-l")))

* 65. Infraestructura FORJA (Dispatch & Macros)

Herramientas centrales del sistema. Deben cargarse antes del hydra de compilación.
- ~my/forja-call-if-loaded~ — reemplaza el patrón =(if (fboundp ...))= repetido en hydras.
- ~my/forja-lang-table~ — dispatch table: los módulos de lenguaje se registran aquí al cargarse.
- ~my/hydra-build~ consulta esta tabla; agregar un lenguaje nuevo = tocar solo su módulo.

#+begin_src emacs-lisp
;;; ── my/forja-call-if-loaded ─────────────────────────────────────────────────

(defmacro my/forja-call-if-loaded (fn &optional msg)
  "Llama FN interactivamente si está definido; muestra MSG si no.
Permite que hydras referencien funciones de módulos opcionales sin errores."
  `(if (fboundp ',fn)
       (call-interactively ',fn)
     (message ,(or msg (format "Módulo no disponible: %s" (symbol-name fn))))))

;;; ── Dispatch Table de Lenguajes ─────────────────────────────────────────────

(defvar my/forja-lang-table (make-hash-table :test #'eq)
  "Registro de lenguajes FORJA.
Clave: símbolo de major-mode.
Valor: plist con :run :run-fallback :build :build-fallback
              :test :test-fallback :format :format-fallback")

(defun my/forja-register-lang (&rest args)
  "Registra un lenguaje en la dispatch table de FORJA.
ARGS es un plist con :modes (lista de major-modes) y acciones opcionales:
  :run :build :test :format  — símbolos de función definidos en el módulo
  :run-fallback :build-fallback :test-fallback :format-fallback — strings para compile"
  (dolist (mode (plist-get args :modes))
    (puthash mode args my/forja-lang-table)))

(defun my/forja-dispatch (action)
  "Ejecuta ACTION para el lenguaje del buffer actual.
ACTION: :run | :build | :test | :format
Prioridad: 1) función del módulo, 2) fallback string, 3) pide al usuario."
  (let* ((entry    (gethash major-mode my/forja-lang-table))
         (fn       (and entry (plist-get entry action)))
         (fb-key   (intern (concat (substring (symbol-name action) 1) "-fallback")))
         (fallback (and entry (plist-get entry fb-key))))
    (cond
     ((and fn (fboundp fn)) (call-interactively fn))
     ((stringp fallback)    (compile fallback))
     (t                     (compile (read-string
                                      (format "%s cmd: " (symbol-name action))))))))
#+end_src

(defun my/git-init-and-commit (project-dir &optional project-name)
  "Inicializa git en PROJECT-DIR y hace el primer commit con todo el scaffolding.
   PROJECT-NAME es opcional, se usa en el mensaje del commit."
  (let ((name (or project-name
                  (file-name-nondirectory (directory-file-name project-dir))))
        (default-directory project-dir))
    (if (not (executable-find "git"))
        (message "⚠️ Git no encontrado en el PATH")
      (let ((result (shell-command-to-string
                     (format
                      "git init && git add -A && git commit -m \"init: scaffolding inicial de %s\""
                      (shell-quote-argument name)))))
        (if (string-match-p "master\\|main" result)
            (message "✅ Git inicializado y primer commit listo en: %s" project-dir)
          (message "⚠️ Algo falló con git: %s" result))))))

;; --- Función Auxiliar: Divide por comas preservando comillas ---
(defun my/split-csv-preserve-quotes (text)
  "Divide TEXT por comas, ignorando las comas dentro de comillas dobles.
   Retorna una lista de elementos limpios."
  (let ((items '())
        (current-item "")
        (in-quote nil)
        (len (length text))
        (i 0))
    (while (< i len)
      (let ((char (aref text i)))
        (cond
         ((char-equal char ?\")
          (setq in-quote (not in-quote))
          (setq current-item (concat current-item (string char))))
         ((and (char-equal char ?,) (not in-quote))
          ;; Si es coma y NO estamos en comillas, guardar item
          (push (string-trim current-item) items)
          (setq current-item ""))
         (t
          (setq current-item (concat current-item (string char))))))
      (setq i (1+ i)))
    ;; Agregar el último elemento
    (push (string-trim current-item) items)
    (nreverse items)))

;; --- Función Principal: Smart Reverse ---
(defun my/smart-reverse ()
  "Rota, transpone o invierte listas CSV de forma inteligente.
   1. Selección: Mueve lo seleccionado al inicio.
   2. Cursor en medio: Transpone el elemento actual con el siguiente.
   3. Cursor al final: Invierte toda la línea."
  (interactive)
  (let* ((separator ", ")
         (eol (line-end-position))
         (bol (line-beginning-position))
         (full-text (buffer-substring-no-properties bol eol))
         (items (my/split-csv-preserve-quotes full-text))
         (new-text nil))

    (cond
     ;; --- CASO 1: Región Seleccionada (Rotar) ---
     ((use-region-p)
      (let* ((beg (region-beginning))
             (end (region-end))
             (sel-text (buffer-substring-no-properties beg end))
             (sel-items (my/split-csv-preserve-quotes sel-text))
             (all-items (my/split-csv-preserve-quotes full-text))
             ;; Filtrar items que NO están en la selección
             (rest-items (cl-remove-if (lambda (x) (member x sel-items)) all-items)))
        (setq new-text (string-join (append sel-items rest-items) separator))))

     ;; --- CASO 2: Cursor al final (Invertir Todo) ---
     ((>= (point) (1- eol))
      (setq new-text (string-join (nreverse items) separator)))

     ;; --- CASO 3: Transponer (Intercambiar vecino) ---
     (t
      (let ((current-len 0)
            (found-index -1)
            (cursor-rel (- (point) bol))) ;; Posición relativa al inicio de línea

        ;; Buscar en qué índice cae el cursor sumando longitudes
        (dotimes (i (length items))
          (let ((item-len (length (nth i items))))
            ;; Sumamos largo del item + 2 (por la coma y espacio)
            (setq current-len (+ current-len item-len 2))
            ;; Si aún no encontramos y el cursor está antes de este límite
            (when (and (< found-index 0) (<= cursor-rel current-len))
              (setq found-index i))))

        ;; Si no encontró (ej: cursor muy al final), asume el último
        (when (< found-index 0) (setq found-index (1- (length items))))

        ;; Lógica de Transposición
        (if (and (>= found-index 0) (< found-index (1- (length items))))
            ;; Si hay un siguiente elemento, intercambiamos A <-> B
            (let ((elem-a (nth found-index items))
                  (elem-b (nth (1+ found-index) items)))
              (let ((rotated-items (append (cl-subseq items 0 found-index)
                                           (list elem-b elem-a)
                                           (cl-subseq items (+ found-index 2)))))
                (setq new-text (string-join rotated-items separator))))

          ;; Si estamos en el último elemento (sin siguiente), invertimos todo por defecto
          (setq new-text (string-join (reverse items) separator))))))

    ;; --- Aplicar ---
    (if new-text
        (progn
          (delete-region bol eol)
          (insert new-text))
      (message "No se pudo procesar la lista."))))

* 95. Sistema de dependencias

Cada módulo de lenguaje registra sus dependencias con ~my/forja-dep~ al cargarse.
~my/forja-warn-missing~ se mantiene como alias para compatibilidad con módulos existentes.
Al terminar el startup, ~my/forja-report-missing~ muestra:
- Solo opcionales → mensaje breve en el echo area.
- Alguna requerida → buffer =*FORJA — Dependencias Faltantes*= con tabla y comandos.

#+begin_src emacs-lisp
(defvar my/forja-missing-deps nil
  "Lista de dependencias faltantes.
Cada entrada es un plist: (:module :binary :type :hint).")

(defun my/forja-dep (&rest args)
  "Registra una dependencia si su binario no está en PATH.
ARGS plist: :module (string) :binary (string) :type (:required|:optional)
            :arch (string|nil) :wsl (string|nil) :termux (string|nil)"
  (let ((module  (plist-get args :module))
        (binary  (plist-get args :binary))
        (type    (or (plist-get args :type) :required))
        (arch    (plist-get args :arch))
        (wsl     (plist-get args :wsl))
        (termux  (plist-get args :termux)))
    (unless (executable-find binary)
      (let ((hint (cond
                    ((bound-and-true-p my/is-termux)
                     (when termux (format "pkg install %s" termux)))
                    ((bound-and-true-p my/is-wsl)
                     (when wsl (format "sudo apt install %s" wsl)))
                    (t
                     (when arch (format "sudo pacman -S %s" arch))))))
        (push (list :module module :binary binary :type type :hint hint)
              my/forja-missing-deps)))))

(defun my/forja-warn-missing (module binary &optional arch-pkg wsl-pkg termux-pkg)
  "Alias de compatibilidad — delega a my/forja-dep con tipo :required."
  (my/forja-dep :module module :binary binary :type :required
                :arch arch-pkg :wsl wsl-pkg :termux termux-pkg))

(defun my/forja-report-missing ()
  "Muestra dependencias faltantes al terminar el startup.
Si hay requeridas abre un buffer detallado; si solo opcionales, mensaje breve."
  (when my/forja-missing-deps
    (let* ((required (cl-remove-if-not
                      (lambda (d) (eq (plist-get d :type) :required))

                      (lambda (d) (eq (plist-get d :type) :optional))
                      my/forja-missing-deps))
           (n-opt (length optional)))
      (if (null required)
          (message "FORJA: %d herramienta%s opcional%s no encontrada%s — M-x my/forja-doctor"
                   n-opt
                   (if (= n-opt 1) "" "s")
                   (if (= n-opt 1) "" "es")
                   (if (= n-opt 1) "" "s"))
        (with-current-buffer (get-buffer-create "*FORJA — Dependencias Faltantes*")
          (read-only-mode -1)
          (erase-buffer)
          (insert (propertize "  FORJA — Dependencias Faltantes\n"
                              'face '(:foreground "#ef4444" :weight bold :height 1.1)))
          (insert (propertize (concat "  " (make-string 50 ?─) "\n\n")
                              'face 'font-lock-comment-face))
          (insert (propertize "  Requeridas:\n" 'face '(:weight bold)))
          (dolist (dep required)
            (insert (format "  ✗  %-22s  [módulo: %s]\n"
                            (plist-get dep :binary)
                            (plist-get dep :module)))
            (when (plist-get dep :hint)
              (insert (propertize (format "       → %s\n" (plist-get dep :hint))
                                  'face '(:foreground "#f59e0b")))))
          (when optional
            (insert "\n")
            (insert (propertize "  Opcionales:\n" 'face '(:weight bold)))
            (dolist (dep optional)
              (insert (format "  ○  %-22s  [módulo: %s]\n"
                              (plist-get dep :binary)
                              (plist-get dep :module)))
              (when (plist-get dep :hint)
                (insert (propertize (format "       → %s\n" (plist-get dep :hint))
                                    'face '(:foreground "#94a3b8"))))))
          (insert "\n")
          (insert (propertize "  M-x my/forja-doctor para diagnóstico completo.\n"
                              'face 'font-lock-comment-face))
          (read-only-mode 1)
          (goto-char (point-min)))
        (display-buffer "*FORJA — Dependencias Faltantes*"
                        '(display-buffer-pop-up-window (window-height . 0.3)))))))
