(defun my/run-live-server ()
  "Lanza live-server npm para el proyecto actual o el directorio del archivo.
En Termux no abre browser (no hay Firefox); en PC abre Firefox."
  (interactive)
  (ignore-errors (kill-process (get-process "live-server-proc")))
  (let ((default-directory
         ;; Usa la raíz del proyecto si existe, sino usa el directorio del archivo actual
         (or (and (bound-and-true-p projectile-mode) (projectile-project-root))
             (file-name-directory buffer-file-name))))
    (if default-directory
        (let ((args (if (bound-and-true-p my/is-termux)
                        '("--quiet" "--no-browser")
                      '("--quiet" "--browser=firefox"))))
          (apply #'start-process "live-server-proc"
                 "*live-server-log*" "live-server" args)
          (message "🚀 Live server iniciado en: %s%s"
                   default-directory
                   (if (bound-and-true-p my/is-termux)
                       " (abrí http://localhost:8080 en el navegador)"
                     "")))
      (message "❌ No se pudo determinar el directorio para live-server"))))

** Compilar y Ejecutar (Node.js)

#+begin_src emacs-lisp
(defun my/node-run ()
  "Ejecuta el proyecto Node.js actual.
Detecta Express/package.json (npm start), o ejecuta el archivo directamente."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           (locate-dominating-file default-directory "package.json")
                           default-directory))
         (default-directory project-root)
         (pkg-json (expand-file-name "package.json" project-root)))
    (cond
     ;; Proyecto con package.json y script "start" o "dev"
     ((file-exists-p pkg-json)
      (let* ((pkg-content (with-temp-buffer
                            (insert-file-contents pkg-json)
                            (buffer-string)))
             (has-dev (string-match-p "\"dev\"" pkg-content))
             (has-start (string-match-p "\"start\"" pkg-content)))
        (cond
         (has-dev   (compile "npm run dev"))
         (has-start (compile "npm start"))
         ;; package.json sin scripts → buscar entry point
         (t (my/node--run-file project-root)))))
     ;; Archivo suelto
     ((buffer-file-name)
      (save-buffer)
      (compile (format "node %s" (file-name-nondirectory buffer-file-name))))
     (t (message "❌ No se encontró archivo JS para ejecutar")))))

(defun my/node--run-file (project-root)
  "Busca y ejecuta el entry point del proyecto Node.js / TypeScript."
  (let ((candidates '("src/index.ts" "src/index.js" "index.ts"
                      "index.js" "app.js" "server.js" "src/app.js")))
    (catch 'found
      (dolist (file candidates)
        (when (file-exists-p (expand-file-name file project-root))
          (compile (if (and (string-match-p "\\.ts$" file) (executable-find "tsx"))
                       (format "tsx %s" file)
                     (format "node %s" file)))
          (throw 'found t)))
      ;; Fallback: ejecutar archivo actual
      (if (buffer-file-name)
          (let ((f (file-name-nondirectory buffer-file-name)))
            (compile (if (and (string-match-p "\\.ts$" f) (executable-find "tsx"))
                         (format "tsx %s" f)
                       (format "node %s" f))))
        (message "❌ No se encontró entry point")))))

(defun my/node-test ()
  "Ejecuta tests del proyecto Node.js.
Detecta jest, mocha, o npm test."
  (interactive)
  (let* ((project-root (or (and (fboundp 'projectile-project-root) (projectile-project-root))
                           (locate-dominating-file default-directory "package.json")
                           default-directory))
         (default-directory project-root))
    (cond
     ;; Jest
     ((file-exists-p (expand-file-name "jest.config.js" project-root))
      (compile "npx jest --verbose"))
     ((file-exists-p (expand-file-name "jest.config.ts" project-root))
      (compile "npx jest --verbose"))
     ;; package.json con script "test"
     ((and (file-exists-p (expand-file-name "package.json" project-root))
           (with-temp-buffer
             (insert-file-contents (expand-file-name "package.json" project-root))
             (string-match-p "\"test\"" (buffer-string))))
      (compile "npm test"))
     ;; Fallback
     (t (compile "npm test")))))

(defun my/node-format-buffer ()
  "Formatea el buffer JS/TS con prettier (si disponible) o lsp."
  (interactive)
  (if (and (fboundp 'prettier-js) (executable-find "prettier"))
      (prettier-js)
    (lsp-format-buffer)))

(defun my/node-keybindings ()
  "Atajos locales para desarrollo en Node.js/Express."
  (local-set-key (kbd "<f5>") #'my/node-run)
  (local-set-key (kbd "C-c C-t") #'my/node-test)
  (local-set-key (kbd "C-c C-f") #'my/node-format-buffer))

;; F5 en JS/TS usa my/node-run (detecta proyecto Node.js)
(add-hook 'js-ts-mode-hook #'my/node-keybindings)
(add-hook 'typescript-ts-mode-hook #'my/node-keybindings)

;; HTML/CSS sigue usando live-server
(add-hook 'html-ts-mode-hook
          (lambda () (local-set-key (kbd "<f5>") 'my/run-live-server)))
(add-hook 'css-ts-mode-hook
          (lambda () (local-set-key (kbd "<f5>") 'my/run-live-server)))

(defun my/new-ts-project (name)
  "Crea un proyecto TypeScript standalone: CLI o librería npm.
Pregunta el tipo con completing-read."
  (interactive "sNombre del proyecto: ")
  (let* ((tipo (completing-read "Tipo de proyecto: " '("cli" "libreria") nil t))
         (base-dir (expand-file-name name (concat (my/mu-projects-base) "ts/")))
         (src-dir  (concat base-dir "/src")))

    (make-directory src-dir t)

    ;; tsconfig.json — base estricto para ambos tipos
    (with-temp-file (concat base-dir "/tsconfig.json")
      (insert "{\n")
      (insert "  \"compilerOptions\": {\n")
      (insert "    \"target\": \"ES2022\",\n")
      (insert "    \"module\": \"commonjs\",\n")
      (insert "    \"lib\": [\"ES2022\"],\n")
      (insert "    \"outDir\": \"./dist\",\n")
      (insert "    \"rootDir\": \"./src\",\n")
      (insert "    \"strict\": true,\n")
      (insert "    \"esModuleInterop\": true,\n")
      (insert "    \"skipLibCheck\": true,\n")
      (insert "    \"declaration\": true,\n")
      (insert "    \"declarationMap\": true,\n")
      (insert "    \"sourceMap\": true\n")
      (insert "  },\n")
      (insert "  \"include\": [\"src/**/*\"],\n")
      (insert "  \"exclude\": [\"node_modules\", \"dist\"]\n")
      (insert "}\n"))

    (if (string= tipo "cli")
        (my/new-ts--cli name base-dir src-dir)
      (my/new-ts--lib name base-dir src-dir))

    ;; .gitignore y .projectile comunes
    (with-temp-file (concat base-dir "/.gitignore")
      (insert "node_modules/\ndist/\n*.js.map\n"))
    (with-temp-file (concat base-dir "/.projectile") (insert ""))

    (my/git-init-and-commit base-dir name)
    (message "Proyecto TypeScript %s creado: %s" tipo base-dir)
    (find-file (concat src-dir "/index.ts"))))

(defun my/new-ts--cli (name base-dir src-dir)
  "Genera template para herramienta CLI en TypeScript."
  ;; package.json con bin y scripts tsx
  (with-temp-file (concat base-dir "/package.json")
    (insert (format "{\n  \"name\": \"%s\",\n" name))
    (insert "  \"version\": \"0.1.0\",\n")
    (insert "  \"description\": \"\",\n")
    (insert "  \"main\": \"dist/index.js\",\n")
    (insert (format "  \"bin\": { \"%s\": \"dist/index.js\" },\n" name))
    (insert "  \"scripts\": {\n")
    (insert "    \"build\": \"tsc\",\n")
    (insert (format "    \"start\": \"tsx src/index.ts\",\n"))
    (insert (format "    \"dev\":   \"tsx watch src/index.ts\",\n"))
    (insert "    \"test\":  \"jest\"\n")
    (insert "  },\n")
    (insert "  \"devDependencies\": {\n")
    (insert "    \"typescript\": \"^5\",\n")
    (insert "    \"@types/node\": \"^20\",\n")
    (insert "    \"tsx\": \"^4\"\n")
    (insert "  }\n")
    (insert "}\n"))
  ;; src/index.ts — entrada CLI con process.argv
  (with-temp-file (concat src-dir "/index.ts")
    (insert "#!/usr/bin/env node\n\n")
    (insert (format "const nombre = \"%s\";\n" name))
    (insert "const args = process.argv.slice(2);\n\n")
    (insert "if (args.length === 0) {\n")
    (insert (format "  console.log(`%s — sin argumentos`);\n" name))
    (insert "  process.exit(0);\n")
    (insert "}\n\n")
    (insert "console.log(`Args: ${args.join(', ')}`);\n")))

(defun my/new-ts--lib (name base-dir src-dir)
  "Genera template para librería npm en TypeScript."
  (with-temp-file (concat base-dir "/package.json")
    (insert (format "{\n  \"name\": \"%s\",\n" name))
    (insert "  \"version\": \"0.1.0\",\n")
    (insert "  \"description\": \"\",\n")
    (insert "  \"main\": \"dist/index.js\",\n")
    (insert "  \"types\": \"dist/index.d.ts\",\n")
    (insert "  \"files\": [\"dist\"],\n")
    (insert "  \"scripts\": {\n")
    (insert "    \"build\": \"tsc\",\n")
    (insert (format "    \"dev\":   \"tsx watch src/index.ts\",\n"))
    (insert "    \"test\":  \"jest\"\n")
    (insert "  },\n")
    (insert "  \"devDependencies\": {\n")
    (insert "    \"typescript\": \"^5\",\n")
    (insert "    \"@types/node\": \"^20\",\n")
    (insert "    \"tsx\": \"^4\"\n")
    (insert "  }\n")
    (insert "}\n"))
  ;; src/index.ts — exports de la librería
  (with-temp-file (concat src-dir "/index.ts")
    (insert (format "// %s — punto de entrada de la librería\n\n" name))
    (insert (format "export function saludo(nombre: string): string {\n"))
    (insert (format "  return `Hola desde %s, ${nombre}!`;\n" name))
    (insert "}\n")))
