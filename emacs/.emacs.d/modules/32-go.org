#+begin_src emacs-lisp
(use-package emacs
  :ensure nil
  :config
  (add-to-list 'treesit-language-source-alist
               '(go "https://github.com/tree-sitter/tree-sitter-go"))
  (add-to-list 'treesit-language-source-alist
               '(gomod "https://github.com/camdencheek/tree-sitter-go-mod"))
  (unless (treesit-language-available-p 'go)
    (condition-case err
        (treesit-install-language-grammar 'go)
      (error (message "⚠️ No se pudo instalar gramática go: %s" err))))
  (unless (treesit-language-available-p 'gomod)
    (condition-case err
        (treesit-install-language-grammar 'gomod)
      (error (message "⚠️ No se pudo instalar gramática gomod: %s" err))))
  ;; go-ts-mode excluido: queries incompatibles con tree-sitter 0.26.x
  ;; TODO: restaurar cuando Emacs actualice queries a sintaxis #match?
  )

#+begin_src emacs-lisp
(defun my/new-go-project (name)
  "Crea un nuevo proyecto Go con go mod init."
  (interactive "sNombre del proyecto: ")
  (unless (executable-find "go")
    (user-error "go no encontrado — instala con: %s"
                (if (bound-and-true-p my/is-wsl) "sudo apt install golang-go" "sudo pacman -S go")))
  (let ((project-dir (expand-file-name name (concat (my/mu-projects-base) "go/"))))
    (make-directory project-dir t)
    (let ((default-directory project-dir))
      (shell-command (format "go mod init %s" name)))
    (with-temp-file (concat project-dir "/main.go")
      (insert "package main\n\n")
      (insert "import \"fmt\"\n\n")
      (insert "func main() {\n")
      (insert "    fmt.Println(\"¡Hola desde Go! 🐹\")\n")
      (insert "}\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert "*.exe\n*.test\n*.out\nvendor/\n"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "🐹 Proyecto Go creado: %s" project-dir)
    (find-file (concat project-dir "/main.go"))))

(defun my/new-go-api-project (name)
  "Crea un nuevo proyecto Go con Gin para REST API."
  (interactive "sNombre del proyecto: ")
  (let ((project-dir (expand-file-name name (concat (my/mu-projects-base) "go/"))))
    (make-directory project-dir t)
    (make-directory (concat project-dir "/handlers") t)
    (make-directory (concat project-dir "/models") t)
    (let ((default-directory project-dir))
      (shell-command (format "go mod init %s" name)))
    (with-temp-file (concat project-dir "/main.go")
      (insert "package main\n\n")
      (insert "import (\n\t\"log\"\n\t\"os\"\n\n")
      (insert (format "\t\"%s/handlers\"\n\n" name))
      (insert "\t\"github.com/gin-contrib/cors\"\n")
      (insert "\t\"github.com/gin-gonic/gin\"\n)\n\n")
      (insert "func main() {\n")
      (insert "\tport := os.Getenv(\"PORT\")\n")
      (insert "\tif port == \"\" {\n\t\tport = \"8080\"\n\t}\n\n")
      (insert "\tr := gin.Default()\n\tr.Use(cors.Default())\n\n")
      (insert "\tr.GET(\"/\", handlers.Index)\n")
      (insert "\tr.GET(\"/health\", handlers.Health)\n\n")
      (insert "\tapi := r.Group(\"/api\")\n\t{\n")
      (insert "\t\tapi.GET(\"/items\", handlers.GetItems)\n")
      (insert "\t\tapi.POST(\"/items\", handlers.CreateItem)\n\t}\n\n")
      (insert (format "\tlog.Printf(\"🐹 Server corriendo en http://localhost:%%s\", port)\n"))
      (insert "\tr.Run(\":\" + port)\n}\n"))
    (with-temp-file (concat project-dir "/handlers/api.go")
      (insert "package handlers\n\n")
      (insert "import (\n\t\"net/http\"\n\t\"github.com/gin-gonic/gin\"\n)\n\n")
      (insert "func Index(c *gin.Context) {\n")
      (insert (format "\tc.JSON(http.StatusOK, gin.H{\"message\": \"Hola desde %s\"})\n}\n\n" name))
      (insert "func Health(c *gin.Context) {\n")
      (insert "\tc.JSON(http.StatusOK, gin.H{\"status\": \"ok\"})\n}\n\n")
      (insert "func GetItems(c *gin.Context) {\n")
      (insert "\tc.JSON(http.StatusOK, gin.H{\"items\": []string{}})\n}\n\n")
      (insert "type CreateItemInput struct {\n")
      (insert "\tName string `json:\"name\" binding:\"required\"`\n}\n\n")
      (insert "func CreateItem(c *gin.Context) {\n")
      (insert "\tvar input CreateItemInput\n")
      (insert "\tif err := c.ShouldBindJSON(&input); err != nil {\n")
      (insert "\t\tc.JSON(http.StatusBadRequest, gin.H{\"error\": err.Error()})\n\t\treturn\n\t}\n")
      (insert "\tc.JSON(http.StatusCreated, gin.H{\"message\": \"Item \" + input.Name + \" creado\"})\n}\n"))
    (with-temp-file (concat project-dir "/.env")
      (insert "PORT=8080\nGIN_MODE=debug\n"))
    (with-temp-file (concat project-dir "/.gitignore")
      (insert (format "*.exe\n*.test\n*.out\n%s\nvendor/\n.env\n" name)))
    (let ((default-directory project-dir))
      (shell-command "go get github.com/gin-gonic/gin github.com/gin-contrib/cors")
      (shell-command "go mod tidy"))
    (with-temp-file (concat project-dir "/.projectile") (insert ""))
    (my/git-init-and-commit project-dir name)
    (message "🐹 Proyecto Go API (Gin) creado: %s" project-dir)
    (find-file (concat project-dir "/main.go"))))
#+end_src
