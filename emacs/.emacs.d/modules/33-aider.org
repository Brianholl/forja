* 15. Autocompletado Inline (Local FIM)

#+begin_src emacs-lisp
;; --- Minuet AI (Autocompletado predictivo tipo fantasma) ---
;; Usa qwen2.5-coder:0.5b offline en Ollama vía openai-fim-compatible.
;; Patcheamos solo los campos necesarios para preservar :template y :get-text-fn.
(use-package minuet
  :ensure t
  :config
  (setq minuet-provider 'openai-fim-compatible)
  (cl-loop for (k v) on
           '(:model      "qwen2.5-coder:0.5b"
             :end-point  "http://localhost:11434/v1/completions"
             :api-key    (lambda () "ollama")
             :name       "Ollama"
             :optional   (:max_tokens 128 :temperature 0.1))
           by #'cddr do
           (setq minuet-openai-fim-compatible-options
                 (plist-put minuet-openai-fim-compatible-options k v)))

  (global-set-key (kbd "M-i") #'minuet-show-suggestion)
  (global-set-key (kbd "M-[") #'minuet-dismiss-suggestion)
  (define-key minuet-active-mode-map (kbd "TAB") #'minuet-accept-suggestion))
#+end_src

* 18. Auto-revert al aplicar cambios

#+begin_src emacs-lisp
;; Cuando Aider aplica cambios a archivos, los buffers abiertos se
;; actualizan automáticamente sin necesidad de recargarlos manualmente.
(defun my/aider--revert-edited-buffer (output)
  "Revierte buffers de archivos editados por Aider al detectar OUTPUT."
  (when (string-match-p "Applied edit to\\|Wrote " output)
    (dolist (buf (buffer-list))
      (with-current-buffer buf
        (when (and buffer-file-name
                   (not (buffer-modified-p))
                   (file-exists-p buffer-file-name))
          (revert-buffer t t t))))))

(defun my/aider--install-revert-hook ()
  "Instala los hooks de output en el buffer de Aider."
  (when-let ((buf (get-buffer (aider-buffer-name))))
    (with-current-buffer buf
      (add-hook 'comint-output-filter-functions
                #'my/aider--revert-edited-buffer nil t)
      (add-hook 'comint-output-filter-functions
                #'my/aider--output-tracker nil t))))

(advice-add 'aider-run-aider :after (lambda (&rest _) (my/aider--install-revert-hook)))
#+end_src

* 19. Indicador visual de actividad del modelo

#+begin_src emacs-lisp
;; Muestra "⚙Aider…" en el modeline mientras el modelo está procesando.
;; Se activa al enviar cualquier comando y se limpia cuando vuelve el prompt "> ".
(defvar my/aider--working nil)

(defun my/aider--output-tracker (output)
  "Limpia el indicador de actividad cuando Aider devuelve el prompt en OUTPUT."
  (when (and my/aider--working (string-match-p "> $" output))
    (setq my/aider--working nil)
    (message "✅ Aider listo")
    (force-mode-line-update t)))

(defun my/aider--mark-working (&rest _)
  "Activa el indicador de actividad al enviar un comando a Aider."
  (setq my/aider--working t)
  (force-mode-line-update t))

(advice-add 'aider-input-sender :before #'my/aider--mark-working)
(advice-add 'aider--comint-send-string-syntax-highlight :before #'my/aider--mark-working)

(add-to-list 'global-mode-string
             '(:eval (when my/aider--working
                       (propertize " ⚙Aider…" 'face 'warning))))
#+end_src

* 20. Funciones de Conveniencia

#+begin_src emacs-lisp
;; --- Verificar si Aider está corriendo ---
(defun my/aider-buffer-running-p ()
  "Verifica si hay un buffer de Aider activo."
  (let ((buf-name (aider-buffer-name)))
    (and buf-name
         (get-buffer buf-name)
         (comint-check-proc buf-name))))

(defun my/aider-open-project ()
  "Abre Aider en la raíz del proyecto actual."
  (interactive)
  (let ((default-directory
         (or (and (fboundp 'projectile-project-root)
                  (projectile-project-root))
             default-directory)))
    (aider-run-aider)
    (message "🤖 Aider iniciado en: %s" default-directory)))

(defun my/aider-add-current-file ()
  "Agrega el archivo actual al contexto de Aider.
   Si Aider no está corriendo, lo inicia primero."
  (interactive)
  (if (buffer-file-name)
      (progn
        (unless (my/aider-buffer-running-p)
          (my/aider-open-project)
          (sit-for 2))
        (aider-add-current-file))
    (message "❌ Este buffer no tiene archivo asociado")))

(defun my/aider-ask-region ()
  "Pregunta a Aider sobre la región seleccionada."
  (interactive)
  (if (use-region-p)
      (aider-send-block-or-region)
    (message "❌ No hay región seleccionada")))

(defun my/aider-fix-flycheck ()
  "Pide a Aider que corrija los errores de Flycheck del buffer actual."
  (interactive)
  (unless (my/aider-buffer-running-p)
    (my/aider-open-project)
    (sit-for 2))
  (my/aider-add-current-file)
  (aider-flycheck-fix-errors-in-scope))

(defun my/aider-keybindings ()
  "Atajos de Aider para modos de programación."
  ;; C-c i abre el hydra — los sub-atajos (o, a, x...) viven dentro del hydra,
  ;; no como secuencias independientes: C-c i no es un prefix keymap.
  (local-set-key (kbd "C-c i") #'my/hydra-aider/body))

(add-hook 'prog-mode-hook #'my/aider-keybindings)
;; Modos que NO heredan de prog-mode → se agregan explícitamente
(add-hook 'web-mode-hook      #'my/aider-keybindings)  ; JSX / Vue
(add-hook 'html-ts-mode-hook  #'my/aider-keybindings)  ; HTML (hereda de text-mode)
#+end_src
